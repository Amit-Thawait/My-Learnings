nginx.yaml
------------------------------
apiVersion: extensions/v1beta1
kind: Deployment
metadata:
  name: my-nginx
spec:
  replicas: 2
  template:
    metadata:
      labels:
        run: my-nginx
    spec:
      containers:
      - name: my-nginx
        image: nginx
        ports:
        - containerPort: 80
------------------------------

kubectl create -f nginx.yaml

vim my-nginx-sv-lb.yaml
-----------------------
apiVersion: v1
kind: Service
metadata:
  name: my-nginx
  labels:
    run: my-nginx
spec:
  type: LoadBalancer
  ports:
  - port: 80
    protocol: TCP
  selector:
    run: my-nginx
-----------------------

kubectl create -f my-nginx-sv-lb.yaml

kubectl describe svc my-nginx
# Note the LoadBalancer IP
# LoadBalancer Ingress: 104.154.52.202
